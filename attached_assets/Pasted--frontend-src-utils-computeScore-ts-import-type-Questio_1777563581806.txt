// frontend/src/utils/computeScore.ts
import type { Question } from "../data/questions.js";
import { QUESTIONS } from "../data/questions.js";

type AnswerValue = string | number | string[];

export type Answer = { questionId: string; value: AnswerValue };

export type DomainScore = { domain: string; score: number; maxScore: number };
export type RiskFactor = { label: string; severity: "low" | "moderate" | "high" };

export type ScoringResult = {
  riskLevel: "low" | "moderate" | "high" | "very_high";
  riskScore: number;
  summary: string;
  recommendations: string[];
  factors: RiskFactor[];
  domainScores: DomainScore[];
  ageRange: string;
};

function getAnswer(answers: Answer[], questionId: string): AnswerValue | undefined {
  return answers.find((a) => a.questionId === questionId)?.value;
}

function asString(v: AnswerValue | undefined): string {
  if (v === undefined) return "";
  if (Array.isArray(v)) return v.join(",");
  return String(v);
}

function asArray(v: AnswerValue | undefined): string[] {
  if (v === undefined) return [];
  if (Array.isArray(v)) return v;
  return [String(v)];
}

export function computeScore(answers: Answer[]): ScoringResult {
  const factors: RiskFactor[] = [];
  const domainTotals: Record<string, { score: number; max: number }> = {
    demographic: { score: 0, max: 0 },
    behavioral: { score: 0, max: 0 },
    epidemiological: { score: 0, max: 0 },
    clinical: { score: 0, max: 0 },
  };

  function add(domain: string, score: number, max: number) {
    domainTotals[domain].score += score;
    domainTotals[domain].max += max;
  }

  // ---------- DEMOGRÁFICO ----------
  const age = asString(getAnswer(answers, "q2_age"));
  const ageRange = age || "no_reportada";
  let demoScore = 0;
  if (age && (age === "18-24" || age === "25-34")) demoScore += 1;
  add("demographic", demoScore, 2);

  const country = asString(getAnswer(answers, "q3_country_birth"));
  const isForeign = country && country.toLowerCase() !== "colombia";

  // ---------- CONDUCTUAL ----------
  const sexWithMen = asString(getAnswer(answers, "q11_sex_with_men"));
  const sexWithWomen = asString(getAnswer(answers, "q12_sex_with_women"));
  const hasSex = sexWithMen === "yes" || sexWithWomen === "yes";

  let menCount = 0, condomMenScore = 0;
  if (sexWithMen === "yes") {
    const countStr = asString(getAnswer(answers, "q11_men_partners_count"));
    if (countStr === "2-5") menCount = 1;
    else if (countStr === "6-10") menCount = 2;
    else if (countStr === "11+") menCount = 3;
    const condom = asString(getAnswer(answers, "q11_condom_use_men"));
    if (condom === "never") condomMenScore = 3;
    else if (condom === "sometimes") condomMenScore = 2;
    else if (condom === "almost_always") condomMenScore = 1;
  }

  let womenCount = 0, condomWomenScore = 0;
  if (sexWithWomen === "yes") {
    const countStr = asString(getAnswer(answers, "q12_women_partners_count"));
    if (countStr === "2-5") womenCount = 1;
    else if (countStr === "6-10") womenCount = 2;
    else if (countStr === "11+") womenCount = 3;
    const condom = asString(getAnswer(answers, "q12_condom_use_women"));
    if (condom === "never") condomWomenScore = 3;
    else if (condom === "sometimes") condomWomenScore = 2;
    else if (condom === "almost_always") condomWomenScore = 1;
  }

  const totalPartnerScore = Math.max(menCount, womenCount);
  if (totalPartnerScore >= 2) {
    factors.push({
      label: "Múltiples parejas sexuales en los últimos 12 meses",
      severity: totalPartnerScore === 3 ? "high" : "moderate",
    });
  }
  add("behavioral", totalPartnerScore, 3);

  const worstCondom = Math.max(condomMenScore, condomWomenScore);
  if (worstCondom >= 2) {
    factors.push({
      label: "Uso inconsistente de condón en los últimos 12 meses",
      severity: worstCondom === 3 ? "high" : "moderate",
    });
  }
  add("behavioral", worstCondom, 3);

  const gender = asString(getAnswer(answers, "q1_gender"));
  const isMale = gender === "man" || gender === "other";
  const isMSM = isMale && sexWithMen === "yes";
  if (isMSM) {
    factors.push({
      label: "Relaciones sexuales entre hombres (HSH) — categoría con mayor incidencia en Colombia",
      severity: "high",
    });
    add("behavioral", 2, 3);
  }

  // ---------- EPIDEMIOLÓGICO ----------
  let epiScore = 0;
  if (isForeign) {
    epiScore += 1;
    factors.push({ label: "Nacido fuera de Colombia", severity: "low" });
  }
  if (asString(getAnswer(answers, "q9_sex_abroad")) === "yes") {
    epiScore += 2;
    factors.push({ label: "Relaciones sexuales con personas fuera de Colombia", severity: "moderate" });
  }
  if (asString(getAnswer(answers, "q10_sex_worker")) === "yes") {
    epiScore += 3;
    factors.push({ label: "Recibir dinero o bienes a cambio de sexo", severity: "high" });
  }
  const injected = asString(getAnswer(answers, "q8_injected_drugs"));
  if (injected === "yes") {
    epiScore += 4;
    factors.push({ label: "Inyección de drogas no prescritas", severity: "high" });
  }

  // --- Contacto reciente con ITS (lógica principal con I=I) ---
  const recentContact = asString(getAnswer(answers, "q6_recent_contact_sti"));
  const recentWhich = asArray(getAnswer(answers, "q6_recent_contact_sti_which"));
  const partnerHiv = asString(getAnswer(answers, "q14_partner_hiv"));

  // Determinar si el contacto reciente incluyó VIH y si esa pareja es la misma con carga indetectable
  const recentContactWithHiv = recentContact === "yes" && recentWhich.includes("hiv");
  const partnerIsUndetectable = partnerHiv === "yes_undetectable";

  let recentContactScore = 0;
  if (recentContact === "yes") {
    if (recentContactWithHiv && partnerIsUndetectable) {
      // No suma riesgo porque I=I (indetectable = intransmisible)
      factors.push({
        label: "Contacto con pareja con VIH pero con carga viral indetectable (I=I) — sin riesgo de transmisión",
        severity: "low",
      });
    } else if (recentContactWithHiv && !partnerIsUndetectable) {
      recentContactScore = 4;
      factors.push({
        label: "Contacto reciente con persona con VIH (estado de tratamiento desconocido o sin alcanzar indetectabilidad)",
        severity: "high",
      });
    } else if (recentWhich.includes("unknown")) {
      recentContactScore = 2;
      factors.push({
        label: "Contacto reciente con persona con ITS (tipo no especificado)",
        severity: "moderate",
      });
    } else if (recentWhich.length > 0) {
      recentContactScore = 3;
      factors.push({
        label: "Contacto reciente con persona con ITS diagnosticada (excluyendo VIH)",
        severity: "high",
      });
    }
  } else if (recentContact === "unsure") {
    recentContactScore = 1;
    factors.push({
      label: "No está seguro/a de haber estado en contacto con ITS",
      severity: "moderate",
    });
  }

  epiScore += recentContactScore;
  add("epidemiological", epiScore, 16); // máximo aumentado para contemplar nuevos puntajes

  // ---------- CLÍNICO ----------
  let clinScore = 0;

  const prep = asString(getAnswer(answers, "q7_prep_use"));
  if (prep === "yes") {
    clinScore -= 2;
    factors.push({ label: "Uso de PrEP (protector)", severity: "low" });
  } else if (prep === "sometimes") clinScore -= 1;

  if (asString(getAnswer(answers, "q5_prior_sti_diagnosis")) === "yes") {
    clinScore += 3;
    factors.push({ label: "Diagnóstico previo de ITS", severity: "high" });
  }

  if (asString(getAnswer(answers, "q4_symptoms")) === "yes") {
    clinScore += 3;
    factors.push({ label: "Síntomas actuales compatibles con ITS", severity: "high" });
  }

  // Pareja con VIH (si ya no fue considerada en contacto reciente por I=I, aquí no se suma)
  if (partnerHiv === "yes_unknown" && !(recentContactWithHiv && partnerIsUndetectable)) {
    clinScore += 4;
    factors.push({ label: "Pareja con VIH sin tratamiento o carga viral desconocida", severity: "high" });
  } else if (partnerHiv === "yes_undetectable") {
    // No suma riesgo, ya se anotó como factor protector antes
    factors.push({ label: "Pareja con VIH en tratamiento (I=I) — no transmisible", severity: "low" });
  } else if (partnerHiv === "dont_know" && hasSex) {
    clinScore += 1;
    factors.push({ label: "Desconocimiento del estado VIH de las parejas", severity: "moderate" });
  }

  const lastTest = asString(getAnswer(answers, "q13_last_hiv_test"));
  if (lastTest === "never" && hasSex) {
    clinScore += 2;
    factors.push({ label: "Nunca se ha realizado la prueba de VIH", severity: "moderate" });
  } else if (lastTest === ">12m" && hasSex) clinScore += 1;
  else if (lastTest === "<3m" && hasSex) clinScore -= 1;

  clinScore = Math.max(0, clinScore);
  add("clinical", clinScore, 14);

  // ---------- PUNTAJE TOTAL ----------
  const totalScore =
    domainTotals.demographic.score +
    domainTotals.behavioral.score +
    domainTotals.epidemiological.score +
    domainTotals.clinical.score;

  let riskLevel: ScoringResult["riskLevel"];
  let summary: string;
  let recommendations: string[];

  if (totalScore <= 2) {
    riskLevel = "low";
    summary = "Riesgo BAJO. Tus prácticas se asocian con baja probabilidad de exposición al VIH.";
    recommendations = [
      "Mantén el uso de condón en todas tus relaciones sexuales.",
      "Realízate una prueba de VIH al menos una vez al año si tienes vida sexual activa.",
      "Conoce y reconoce los síntomas de ITS y consulta si aparecen.",
    ];
  } else if (totalScore <= 5) {
    riskLevel = "moderate";
    summary = "Riesgo MODERADO. Algunas situaciones requieren atención preventiva.";
    recommendations = [
      "Realízate una prueba de VIH y otras ITS en los próximos 1 a 3 meses.",
      "Refuerza el uso correcto y consistente del condón.",
      "Conversa con un profesional de salud sobre la PrEP.",
    ];
  } else if (totalScore <= 9) {
    riskLevel = "high";
    summary = "Riesgo ALTO. Actúa pronto para cuidar tu salud.";
    recommendations = [
      "Acude a un centro de salud en Manizales para una prueba de VIH y otras ITS lo antes posible.",
      "Pregunta por la PrEP — es altamente eficaz.",
      "Si tuviste una exposición de riesgo en las últimas 72 horas, consulta por PEP.",
    ];
  } else {
    riskLevel = "very_high";
    summary = "Riesgo MUY ALTO. Realízate una prueba de VIH cuanto antes.";
    recommendations = [
      "Realízate una prueba de VIH en los próximos días — muchas son gratuitas en Manizales.",
      "Consulta urgentemente por PEP si exposición reciente.",
      "Considera iniciar PrEP.",
      "Recuerda: el diagnóstico temprano permite vivir con VIH una vida plena y saludable.",
    ];
  }

  if (asString(getAnswer(answers, "q4_symptoms")) === "yes") {
    recommendations.unshift("🔴 Presentas síntomas compatibles con ITS. ¡Acude a un centro de salud de inmediato!");
  }

  const domainScores: DomainScore[] = [
    { domain: "Conductual", score: domainTotals.behavioral.score, maxScore: domainTotals.behavioral.max },
    { domain: "Epidemiológico", score: domainTotals.epidemiological.score, maxScore: domainTotals.epidemiological.max },
    { domain: "Clínico", score: domainTotals.clinical.score, maxScore: domainTotals.clinical.max },
    { domain: "Demográfico", score: domainTotals.demographic.score, maxScore: domainTotals.demographic.max },
  ];

  return {
    riskLevel,
    riskScore: totalScore,
    summary,
    recommendations,
    factors,
    domainScores,
    ageRange,
  };
}
